1пјүCREATE TABLEиҜӯеҸҘжҺҘеҸ—FULLTEXTеӯҗеҸҘпјҢе®ғз»ҷеҮәиў«зҙўеј•еҲ—зҡ„дёҖдёӘйҖ—еҸ·еҲҶйҡ”зҡ„еҲ—иЎЁ

2пјүеңЁзҙўеј•д№ӢеҗҺпјҢдҪҝз”ЁдёӨдёӘеҮҪж•°Match()е’ҢAgainst()жү§иЎҢе…Ёж–Үжң¬жҗңзҙўпјҢе…¶дёӯMatch()жҢҮе®ҡиў«жҗңзҙўзҡ„еҲ—пјҢAgainst()жҢҮе®ҡиҰҒдҪҝз”Ёзҡ„жҗңзҙўиЎЁиҫҫејҸпјӣдј йҖ’з»ҷMatch()зҡ„еҖјеҝ…йЎ»дёҺFULLTEXT()е®ҡд№үдёӯзҡ„зӣёеҗҢпјҢеҰӮжһңжҢҮе®ҡеӨҡдёӘеҲ—пјҢеҲҷеҝ…йЎ»еҲ—еҮәе®ғд»¬пјҲиҖҢдё”ж¬ЎеәҸжӯЈзЎ®пјү
SELECT note_text
FROM productnotes
WHERE Match(note_text) Against(вҖҳrabbitвҖҷ);

3пјүиҝҷжқЎSELECTиҜӯеҸҘеҗҢж ·жЈҖзҙўеҮәдёӨиЎҢпјҢдҪҶж¬ЎеәҸдёҚеҗҢпјҲиҷҪз„¶е№¶дёҚжҖ»жҳҜеҮәзҺ°иҝҷз§Қжғ…еҶөпјү
SELECT note_text
FROM productnotes
WHERE note_text LIKE вҖҳ%rabbit%вҖҷ;

дёҠиҝ°дёӨжқЎSELECTиҜӯеҸҘйғҪдёҚеҢ…еҗ«ORDER BYеӯҗеҸҘпјҢеҗҺиҖ…д»ҘдёҚзү№еҲ«жңүз”Ёзҡ„йЎәеәҸиҝ”еӣһж•°жҚ®гҖӮеүҚиҖ…иҝ”еӣһд»Ҙж–Үжң¬еҢ№й…Қзҡ„иүҜеҘҪзЁӢеәҰжҺ’еәҸзҡ„ж•°жҚ®гҖӮдёӨдёӘиЎҢйғҪеҢ…еҗ«иҜҚrabbitпјҢдҪҶеҢ…еҗ«иҜҚrabbitдҪңдёәз¬¬3дёӘиҜҚзҡ„иЎҢзҡ„зӯүзә§жҜ”дҪңдёәз¬¬20дёӘиҜҚзҡ„иЎҢй«ҳ

4пјүжј”зӨәжҺ’еәҸеҰӮдҪ•е·ҘдҪң
SELECT note_textпјҢ
Match(note_text) Against(вҖҳrabbitвҖҷ) AS rank
